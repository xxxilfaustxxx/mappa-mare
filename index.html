// --- GESTIONE WAYPOINTS (Click Destro per creare, Doppio Click per eliminare) --- map.on('contextmenu', function(e) { // Creiamo il waypoint var waypoint = L.marker(e.latlng, { draggable: true, title: "Doppio click per eliminare" }).addTo(map); // Creiamo il contenuto del popup var popupContent = "Punto Nave
" + "Lat: " + e.latlng.lat.toFixed(5) + "
" + "Lon: " + e.latlng.lng.toFixed(5) + "
" + "Doppio click sul punto per rimuoverlo"; waypoint.bindPopup(popupContent); // EVENTO ELIMINAZIONE: Se fai doppio click sul marker, sparisce waypoint.on('dblclick', function(e) { map.removeLayer(waypoint); }); // Se trascini il punto, aggiorna le coordinate nel popup waypoint.on('dragend', function(e) { var newPos = e.target.getLatLng(); e.target.setPopupContent( "Punto Nave (Spostato)
" + "Lat: " + newPos.lat.toFixed(5) + "
" + "Lon: " + newPos.lng.toFixed(5) + "
" + "Doppio click per rimuovere" ); }); });
